Python
import asyncio

async def cultivate(level, duration):
    print(f"开始修炼 {level}...")
    await asyncio.sleep(duration)
    print(f"✅ {level} 修炼完成！")

# 三种异火并发燃烧
asyncio.run(asyncio.gather(
    cultivate("青莲地心火", 1),
    cultivate("陨落心炎", 2),
    cultivate("三千焱炎火", 1.5),
))
JavaScript
// 焚诀进化判定
function canEvolve(currentRank, flameRank) {
  const req = {
    '黄阶': '玄阶', '玄阶': '地阶',
    '地阶': '天阶', '天阶': '异火'
  };
  return flameRank >= req[currentRank];
}

// 测试
console.log(canEvolve('玄阶', '地阶')); // true
console.log(canEvolve('天阶', '地阶')); // false

      const originalText = btnElement.textContent.replace('已复制！', '').trim() || "复制";
      
      // 3. 写入剪贴板
      navigator.clipboard.writeText(text).then(() => {
        // 变成成功状态
        btnElement.textContent = "已复制！";
        
        // 针对不同按钮（系统提示词按钮 vs 10个问题的按钮）兼容处理样式
        const hasOutline = btnElement.classList.contains("btn-outline-primary");
        if (hasOutline) {
          btnElement.classList.remove("btn-outline-primary");
          btnElement.classList.add("btn-success");
        } else {
          // 针对顶部的系统提示词原生样式按钮
          btnElement.style.backgroundColor = "#4caf50";
        }
        
        // 4. 【核心修复】1.5秒后必强制恢复，绝不卡死
        setTimeout(() => {
          btnElement.textContent = originalText;
          if (hasOutline) {
            btnElement.classList.remove("btn-success");
            btnElement.classList.add("btn-outline-primary");
          } else {
            btnElement.style.backgroundColor = ""; // 恢复 CSS 原生 hover 颜色
          }
        }, 1500);
        
      }).catch(err => {
        console.error("复制失败:", err);
        alert("浏览器限制了自动复制，请手动选择并复制文本。");
      });
    }
  </script>
